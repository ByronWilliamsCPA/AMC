# 3.3 Responsive/Mobile + Accessibility Pass Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a full-stack Playwright e2e harness that audits the AMC React SPA for accessibility (WCAG 2.1 AA) and mobile responsiveness, fix the real gaps the audit surfaces, lock each fix with a regression assertion, gate it in CI, and document the result.

**Architecture:** The e2e suite runs the real FastAPI app on a throwaway on-disk SQLite database seeded from a committed *synthetic* `content/e2e_seed.json` (never the real `content/amc_data.json`). Playwright builds the production frontend bundle and serves it via `vite preview`, with a new `preview.proxy` block forwarding `/api` and `/health` to the API on `:8000` so the `amc_session` cookie is same-origin. A `globalSetup` bootstraps a coach (out-of-band), mints a student invite, registers and logs in the student, and saves `storageState` so specs start authenticated. Audit specs (axe, responsive overflow, keyboard, Lighthouse) are the measuring instrument; remediation is driven by what they report, not pre-baked.

**Tech Stack:** Playwright (`@playwright/test`), `@axe-core/playwright`, `playwright-lighthouse` + `lighthouse`, Vite 6 preview, FastAPI + uvicorn, Alembic, SQLite (`aiosqlite`), GitHub Actions.

depends-on: none

- [ ] **Step 1: Write the generator**

Create `scripts/e2e/build_e2e_seed.py` with the exact content below. It emits a single file containing BOTH the AMC keys (`tests`, `byContest`, `keyedTests`) and the DIAG keys (`instruments`, `order`, `ladder`, `catalog`), so `--amc <file> --diag <file>` and the two-arg validator both work.

```python
#!/usr/bin/env python3
"""Generate the synthetic e2e seed used by the Playwright harness.

The output contains no real AMC problem text or images: it is 25 trivial
LaTeX problems (one deliberately wide, to exercise KaTeX horizontal overflow)
plus one minimal diagnostic instrument and catalog row. It passes
``content/validate_content.py`` and seeds via ``python -m amc.seed``.

Run from the repo root:

    uv run python scripts/e2e/build_e2e_seed.py
"""

from __future__ import annotations

import json
from pathlib import Path

# Problem 13 is intentionally wide to exercise KaTeX overflow at 320-375px.
_WIDE_LATEX = (
    r"\sum_{k=1}^{40} \left( a_k + b_k + c_k + d_k + e_k + f_k + g_k "
    r"+ h_k + i_k + j_k \right)^2 = \prod_{m=1}^{40} (x_m - y_m)(p_m - q_m)"
)


def _problem(n: int) -> dict[str, object]:
    q = _WIDE_LATEX if n == 13 else f"Synthetic problem {n}: compute {n} + {n}."
    return {
        "n": n,
        "q": q,
        "choices": [{"L": letter, "html": f"{letter}. choice {n}"} for letter in "ABCDE"],
        "sol": f"https://example.test/solution/{n}",
    }


def build() -> dict[str, object]:
    answers = [["A", "B", "C", "D", "E"][n % 5] for n in range(25)]
    return {
        "tests": {
            "AMC10-2099": {
                "id": "AMC10-2099",
                "contest": "AMC 10",
                "year": 2099,
                "exam": "",
                "durationSec": 4500,
                "scoreMode": "count",
                "mode": "latex",
                "voided": [],
                "answers": answers,
                "problems": [_problem(n) for n in range(1, 26)],
            }
        },
        "byContest": {"AMC 8": [], "AMC 10": ["AMC10-2099"], "AMC 12": []},
        "keyedTests": [],
        "instruments": {
            "e2e-diag": {
                "id": "e2e-diag",
                "course": "E2E Synthetic Course",
                "role": "AYR",
                "kind": "",
                "grading": {"mode": "single", "total": 2, "need": 1},
                "instructions": "Synthetic diagnostic for e2e.",
                "sections": [
                    {
                        "title": "Section 1",
                        "items": [
                            {
                                "id": "e2e-1",
                                "label": "1",
                                "prompt": "What is 2 + 2?",
                                "ans": "4",
                                "v": 4,
                                "accept": ["4"],
                                "manual": False,
                            },
                            {
                                "id": "e2e-2",
                                "label": "2",
                                "prompt": "Explain your reasoning.",
                                "ans": "free response",
                                "manual": True,
                            },
                        ],
                    }
                ],
            }
        },
        "order": ["e2e-diag"],
        "ladder": [],
        "catalog": [
            {"course": "E2E Synthetic Course", "gate": "diagnostic", "note": "e2e"}
        ],
    }


def main() -> None:
    out = Path(__file__).resolve().parents[2] / "content" / "e2e_seed.json"
    out.write_text(json.dumps(build(), indent=2) + "\n", encoding="utf-8")
    print(f"wrote {out}")  # noqa: T201


if __name__ == "__main__":
    main()
```

### Task A8: Auth bootstrap `frontend/e2e/global-setup.ts`

depends-on: A7 [output]

**Files:**
- Create: `frontend/e2e/global-setup.ts`

**Context:** All auth calls go through the preview origin (`http://localhost:3000`), NOT directly to `:8000`. The `amc_session` cookie set by the API has no `Domain` attribute, so routing the login through the proxy scopes the saved cookie to `localhost:3000` (the origin the specs run against). Coach work uses one request context; the student a fresh one, so the saved state holds only the student cookie.

- [ ] **Step 1: Write the global setup**

Create `frontend/e2e/global-setup.ts` with exactly:

```ts
import { mkdirSync } from 'node:fs'
import path from 'node:path'
import { request, type FullConfig } from '@playwright/test'

const BASE = 'http://localhost:3000'
const STUDENT_STATE = path.join(__dirname, '.auth', 'student.json')

// Synthetic, throwaway e2e credentials (must match scripts/e2e/run-api.sh);
// never real secrets. The allowlist pragmas keep detect-secrets from flagging
// these test fixtures here and in the real file.
const COACH = { email: 'coach@example.test', password: 'e2e-coach-pw-12345' } // pragma: allowlist secret
const STUDENT = {
  email: 'student@example.test',
  display_name: 'E2E Student',
  password: 'e2e-student-pw-12345', // pragma: allowlist secret
}

async function globalSetup(_config: FullConfig): Promise<void> {
  mkdirSync(path.dirname(STUDENT_STATE), { recursive: true })

  // 1. Coach context: log in, mint a student invite.
  const coachCtx = await request.newContext({ baseURL: BASE })
  const coachLogin = await coachCtx.post('/api/v1/auth/login', { data: COACH })
  if (!coachLogin.ok()) {
    throw new Error(`coach login failed: ${coachLogin.status()} ${await coachLogin.text()}`)
  }
  const inviteRes = await coachCtx.post('/api/v1/invites', {
    data: { email: STUDENT.email, role: 'student' },
  })
  if (inviteRes.status() !== 201) {
    throw new Error(`invite mint failed: ${inviteRes.status()} ${await inviteRes.text()}`)
  }
  const { token } = (await inviteRes.json()) as { token: string }
  await coachCtx.dispose()

  // 2. Student context: register with the invite, then log in.
  const studentCtx = await request.newContext({ baseURL: BASE })
  const register = await studentCtx.post('/api/v1/auth/register', {
    data: { token, display_name: STUDENT.display_name, password: STUDENT.password },
  })
  if (register.status() !== 201) {
    throw new Error(`register failed: ${register.status()} ${await register.text()}`)
  }
  const studentLogin = await studentCtx.post('/api/v1/auth/login', {
    data: { email: STUDENT.email, password: STUDENT.password },
  })
  if (!studentLogin.ok()) {
    throw new Error(`student login failed: ${studentLogin.status()} ${await studentLogin.text()}`)
  }

  // 3. Save the student cookie for specs.
  await studentCtx.storageState({ path: STUDENT_STATE })
  await studentCtx.dispose()
}

export default globalSetup
```

/** Assert the document has no horizontal overflow (allow 1px rounding). */
export async function expectNoHorizontalOverflow(page: Page): Promise<void> {
  const overflow = await page.evaluate(() => {
    const el = document.documentElement
    return { scroll: el.scrollWidth, client: el.clientWidth }
  })
  expect(
    overflow.scroll,
    `horizontal overflow: scrollWidth ${overflow.scroll} > clientWidth ${overflow.client}`,
  ).toBeLessThanOrEqual(overflow.client + 1)
}
